using System;

namespace BasicClass // namespace로 BasicClass이름으로 생성
{
	class Cat  // Cat 클래스 생성
	{
		public string Name; // string형 변수 Name 생성
		public string Color; // string형 변수 Color 생성

		public void Meow() // 함수 Meow 생성
		{ 
			Console.WriteLine($"{Name}:  야옹"); // "Name값 : 야옹" 출력
		}
	}

	class MainApp { // MainApp 클래스 생성
		static void Main(string[] args)  // 매개변수 args를 가지고 있는 Main 함수 생성
		{
			Cat kitty = new Cat(); //  kitty를 Cat클래스로 만든다 
			kitty.Color = "하얀색"; // kitty의 Color 값에  "하얀색"을 넣는다
			kitty.Name = " 키티 "; // kitty의 Name 값에  "키티"를 넣는다
			kitty.Meow(); // kitty의  Meow함수 를 호출
			Console.WriteLine($"{kitty.Name}: {kitty.Color}"); // "kitty.Name값 : kitty.Color값" 출력

			Cat nero = new Cat();  nero를 Cat클래스로 만든다 
			nero.Color = "검은색"; // nero의 Color 값에  "검은색"을 넣는다
			nero.Name = " 네로 "; // nero의 Name 값에  "네로"를 넣는다
			nero.Meow(); // nero의  Meow함수 를 호출
			Console.WriteLine($"{nero.Name}: {nero.Color}");// "nero.Name값 : nero.Color값" 출력
		}
	}
}


using System;

namespace Constructor // namespace로 Constructor이름으로 생성
{
	class Cat // Cat 클래스 생성
	{
		public Cat() 
		{
			Name = ""; // Name에 "" 입력 
			Color = ""; // Color에 ""입력
		}

		public Cat(string _Name, string _Color)
		{
			Name = _Name;  // Name에 _Name 입력
			Color = _Color;   // Color에 _Color입력
		}

		~Cat()
		{
			Console.WriteLine($"{Name}: 잘가"); // "Name값 : 잘가" 출력
		}
	}

	class MainApp // MainApp class 입력
	{
		static void Main(string[] args)  // string args를 포함한 Main 함수를 생성
		{
			Cat kitty = new Cat("키티", "하얀색"); // kitty를 Cat 함수로 바꾼다
			kitty.Meow(); // kitty.Meow() 호출
			Console.WriteLine($"{kitty.Name}:{kitty.Color}"); // "kitty.Name값:kitty.Colorr값" 출력

			Cat nero = new Cat("네로", "검은색"); / nero를 Cat 함수로 바꾼다
			nero.Meow(); nero.Meow 호출
			Console.WriteLine($"{nero.Name) : {nero.Color}");"nero.Name값:nero.Colorr값" 출력
		}
	}

}
